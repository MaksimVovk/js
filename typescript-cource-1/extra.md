# Watch mode

> tsc app.ts -w

# tsconfig.json file

> tsc --init - *init "tsconfig.json" file*
> tsc -w - *wach mode for all file in a project*

### "exclude"
Exclude file or directory

> "exclude": ["./Basics", "app.ts"]

### "include"
Include file or directory

> "include": ["./Basics", "app.ts"]
